Листинг 1. Алгоритм получения максимально возможного уровня освоения семантических квантов.

function getBestResult(setQuants, totalTime)
    INPUT: Структура данных setQuants представляет набор семантических квантов
           Структура объекта setQuants = { level: { quantum: { time, score }, ..., quantum_n: {...} }, ...},
           где level - уровень освоения, quantum - квант, time - время на освоение, score - балл для заданного уровня
           Общее допустимое время на освоение семантических квантов - totalTime.
    OUTPUT: Объект result содержит максимально возможный уровень освоения семантических квантов
            и наилучшее распределение времени

    result := array([])
    arr := array([totalTime + 1])
    prevArr := array([])

    for level in setQuants
      for quant in level
        for index in arr

          if quant.time <= index then
            leftTime := index - quant.time

            if leftTime !== 0 && prevArr.length !== 0 then
              if prevArr[leftTime].includes(quant) then
                if prevArr[index].max > quant.score + prevArr[leftTime].max - prevArr[leftTime][quant].score then
                  arr[index] = prevArr[index]
                else
                  arr[index] = {...prevArr[leftTime], max:quant.score + prevArr[leftTime].max - prevArr[leftTime][quant].score, [quant]: { keyLevel, time: quant.time, score: quant.score}}
                end if
              else 
                if prevArr[index].max > quant.score + prevArr[leftTime].max then
                  arr[index] = prevArr[index]
                else
                  arr[index] = {...prevArr[leftTime], max: quant.score + prevArr[leftTime].max, [quant]: { keyLevel, time: quant.time, score: quant.score}}
                end if
              end if
            else
              if prevArr.length && prevArr[index].max > quant.score
                arr[index] = prevArr[index]
              else
                arr[index] = { max: quant.score, [quant]: { keyLevel, time: quant.time, score: quant.score}}
              end if
            end if
          end if
        end for

        result.push(arr[arr.length - 1])
        prevArr := arr

      end for
    end for

    if quant.time <= index then
      return result[result.length - 1]
    else 
      return 'It is impossible to master every quantum in' + totalTime + 'units of time'
    end if
end function
